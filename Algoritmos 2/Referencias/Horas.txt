int diferenciaH(int &hora1, int &hora2);

int diferenciaH(int &hora1, int &hora2){
    int hf;
    if (hora2 >= hora1){
        hf = hora2-hora1;
    }else{
        hf = hora1-hora2;
    }
    return hf;
}
int diferenciaM(int &min1, int &min2){
    int mf;
    if (min2 >= min1){
        mf = min2-min1;
    }else{
        mf = min1-min2;
    }
    return mf;
}
